Структура объекта товара
{
  "id": "", - идентификатор товара
  "active": false, - для selected товара true, для оставльных всегда false
  "name": "" - наименование товара
  "height": "", - высота
  "width": "", - ширина
  "length": "", - длинна
  "image": "", - url адрес основной картинки товара
  "preview": "", - url адрес превьюшки для слайдера
  "link": "" - ссылка на страницу товара
}

Структура объекта картинки интерьера
{
  "id": "1", - идентификатор картинки
  "active": true, - если нужно чтобы конкретный интерьер был выбран. По умолчанию всегда ставим превому
  "image": "", - url адрес большой картинки
  "preview": "" - url адрес превьюшки для слайдера
}

Структура json файла с данными
{
  "selected": { объект товара с данными }, - содержит только один объект товара. Если пользователь не выбрал товар для примерки, selected должен отдаваться либо со значением null, либо вообще его не добавлять

  "viewed": [ - массив с объектами типа товар
    { объект товара },
    ...
  ],

  "sketches": [ - массив с объектами типа картинка интерьера
    { объект картинки интерьера },
    ...
  ],
  "backlink": "" - url адрес сылки для перехода СО СТРАНИЦЫ примерки. Может быть сслыкой на каталог, главну, или предыдущую страницу с которой пришел пользователь.
}
